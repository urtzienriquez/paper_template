\newcommand{\titleblock}{
    \begin{center}
        % title
        \LARGE $title$

        \vspace{1em}
        % author list
        \large
        $for(author)$$author.name$$if(author.affiliation)$\textsuperscript{$author.affiliation$}$endif$$if(author.corresponding)$\textsuperscript{*}$endif$$sep$,\space
$endfor$

        \vspace{0.5em}
        % affiliations
        \small
        $for(affiliations)$\textsuperscript{$affiliations.id$}$affiliations.text$ \par
        $endfor$
        % corresponding author
        $for(author)$$if(author.corresponding)$\textsuperscript{*}Corresponding author: $author.email$$endif$$endfor$
    \end{center}
    \normalsize
}
